dice_p <- function(w,t=3,s=10){
  
  w # 1
  v <- 1-w # 1
  t # 
  s # 
  
  num <- prod(seq.int(s,s-t+1))
  den <- prod(seq.int(t:1))
  const <- num/den # 
  
  # 
  res <- w^t * v^(s-t)* const # probability 
  return(res)
}

dice_b <- function(w,t=3,s=10,a=2,b=5){
  
  w # 1
  v <- 1-w # 1
  
  res <- w^t * v^(s-t) * dbeta(w,a,b)# posterior  likelihood * prior
  return(res)
}
